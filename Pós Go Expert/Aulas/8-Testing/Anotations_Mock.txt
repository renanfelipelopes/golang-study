// =====================================================================================
// 📦 MOCKS EM GO (GUIA PRÁTICO E DIDÁTICO)
// =====================================================================================
//
// Mock é uma implementação falsa de uma dependência usada em testes.
//
// 🎯 Objetivo:
// - Isolar a unidade de código sendo testada
// - Evitar dependências reais (DB, API, filesystem, etc)
// - Controlar comportamento (sucesso/erro)
// - Validar interações (quantas vezes foi chamado, com quais parâmetros)
//
// 💡 Em Go, usamos muito:
// - Interfaces + Injeção de Dependência (DI)
// - Bibliotecas como: testify/mock
//
// =====================================================================================
// 🧠 CONCEITO-CHAVE: DEPENDENCY INJECTION (DI)
// =====================================================================================
//
// Em vez de criar dependências dentro da função,
// nós recebemos elas como parâmetro.
//
// Isso permite substituir por mocks em testes.
//
// =====================================================================================
// 📄 tax.go (EXPLICADO)
// =====================================================================================

package tax

import "errors"

// Interface define um contrato
// 💡 Isso permite trocar implementação real por mock
type Repository interface {
	SaveTax(amount float64) error
}

// Função que depende de um repositório
// 🔥 Aqui está o ponto-chave para uso de mocks
func CalculateTaxAndSave(amount float64, repository Repository) error {

	// Calcula imposto
	tax := CalculateTax2(amount)

	// Salva via interface (pode ser mock ou implementação real)
	return repository.SaveTax(tax)
}

// Função pura (sem efeitos colaterais)
func CalculateTax2(amount float64) float64 {
	if amount <= 0 {
		return 0
	}
	if amount >= 1000 && amount < 20000 {
		return 10.0
	}
	if amount >= 20000 {
		return 20.0
	}
	return 5.0
}

// Versão com erro
func CalculateTax(amount float64) (float64, error) {
	if amount <= 0 {
		return 0, errors.New("Amount must be greater than 0")
	}
	if amount >= 1000 && amount < 20000 {
		return 10.0, nil
	}
	if amount >= 20000 {
		return 20.0, nil
	}
	return 5.0, nil
}

package tax

import (
	"github.com/stretchr/testify/mock"
)

// Mock do repositório
// 🔥 Embedding de mock.Mock (herança "composição" em Go)
type TaxRepositoryMock struct {
	mock.Mock
}

// Implementação fake do método
func (m *TaxRepositoryMock) SaveTax(tax float64) error {

	// 🔍 m.Called registra a chamada e busca comportamento configurado
	args := m.Called(tax)

	// Retorna erro configurado no teste
	return args.Error(0)
}

// 📄 tax_test.go (EXPLICADO)
// =====================================================================================

package tax

import (
	"errors"
	"testing"

	"github.com/stretchr/testify/assert"
)

func TestCalculateTax(t *testing.T) {

	// Caso válido
	tax, err := CalculateTax(1000.0)

	// assert.Nil → verifica se err é nil
	assert.Nil(t, err)

	// assert.Equal → compara valores
	assert.Equal(t, 10.0, tax)

	// Caso inválido
	tax, err = CalculateTax(0)

	// assert.Error → verifica se houve erro
	assert.Error(t, err, "Amount must be greater than 0")

	// Mesmo com erro, validamos retorno
	assert.Equal(t, 0.0, tax)

	// Verifica conteúdo da mensagem
	assert.Contains(t, err.Error(), "greater than 0")
}

// 🔥 TESTE COM MOCK (PARTE MAIS IMPORTANTE)
func TestCalculateTax2(t *testing.T) {

	// Criação do mock
	repository := &TaxRepositoryMock{}

	// =================================================================================
	// 🎯 DEFINIÇÃO DE COMPORTAMENTO (EXPECTATIONS)
	// =================================================================================
	//
	// Quando SaveTax for chamado com 10.0 → retorna nil
	repository.On("SaveTax", 10.0).Return(nil)

	// Quando SaveTax for chamado com 0.0 → retorna erro
	repository.On("SaveTax", 0.0).Return(errors.New("error saving tax"))

	// 💡 Alternativas poderosas:
	//
	// Qualquer valor:
	// repository.On("SaveTax", mock.Anything).Return(errors.New("error"))
	//
	// Controlar quantidade de chamadas:
	// .Once() / .Twice()
	//
	// repository.On("SaveTax", 10.0).Return(nil).Once()

	// =================================================================================
	// 🚀 EXECUÇÃO
	// =================================================================================

	// Caso sucesso
	err := CalculateTaxAndSave(1000.0, repository)
	assert.Nil(t, err)

	// Caso erro
	err = CalculateTaxAndSave(0.0, repository)
	assert.Error(t, err, "error saving tax")

	// =================================================================================
	// 🔍 VALIDAÇÕES DE INTERAÇÃO
	// =================================================================================

	// Verifica se TODAS expectativas foram atendidas
	repository.AssertExpectations(t)

	// Verifica número de chamadas
	//
	// ⚠️ IMPORTANTE:
	// Aqui você colocou 3 chamadas
	//
	// Vamos analisar:
	//
	// CalculateTaxAndSave(1000) → SaveTax(10)
	// CalculateTaxAndSave(0)    → SaveTax(0)
	//
	// 👉 TOTAL: 2 chamadas
	//
	// ❌ AssertNumberOfCalls(t, "SaveTax", 3) provavelmente vai falhar
	//
	repository.AssertNumberOfCalls(t, "SaveTax", 3)
}

// =====================================================================================
// 🧠 CONCEITOS ESSENCIAIS (RESUMO)
// =====================================================================================
//
// MOCK:
// - Implementação fake
// - Controla comportamento
// - Permite validar chamadas
//
// INTERFACE:
// - Permite desacoplamento
// - Essencial para mockar em Go
//
// TESTIFY:
// - Facilita asserts e mocks
//
// =====================================================================================
// 📦 EXEMPLOS ADICIONAIS
// =====================================================================================

// Exemplo corrigindo número de chamadas
func TestCalculateTax2Fixed(t *testing.T) {
	repository := &TaxRepositoryMock{}

	repository.On("SaveTax", 10.0).Return(nil)
	repository.On("SaveTax", 0.0).Return(errors.New("error saving tax"))

	CalculateTaxAndSave(1000.0, repository)
	CalculateTaxAndSave(0.0, repository)

	// ✔️ Correto: 2 chamadas
	repository.AssertNumberOfCalls(t, "SaveTax", 2)
}

// Exemplo usando mock.Anything
func TestCalculateTaxWithAny(t *testing.T) {
	repository := &TaxRepositoryMock{}

	repository.On("SaveTax", mock.Anything).Return(nil)

	err := CalculateTaxAndSave(500.0, repository)

	assert.Nil(t, err)
	repository.AssertExpectations(t)
}

// =====================================================================================
// ✅ BOAS PRÁTICAS
// =====================================================================================
//
// - Usar interfaces para dependências
// - Mockar apenas dependências externas (DB, API)
// - Testar comportamento, não implementação
// - Validar interações importantes
// - Evitar over-mocking
//
// =====================================================================================
// ⚠️ ARMADILHAS COMUNS
// =====================================================================================
//
// - Mockar tudo (anti-pattern)
// - Não validar chamadas
// - Usar valores muito específicos (testes frágeis)
// - Ignorar AssertExpectations
// - Contar chamadas errado (como no seu exemplo)
//
